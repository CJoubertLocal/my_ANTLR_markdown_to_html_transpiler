```Pseudocode
fn removeCharacterFromList(remList list, charToRemove char) list {
    match remList {
        case x::[]:
            match x {
                charToRemove: []
                _: x
            }
        case x::xs:
            match x {
                charToRemove: removeCharacterFromList(xs, charToRemove)
                _: x::removeCharacterFromList(xs, charToRemove)
            }
    }
}

removeCharacterFromList(['a', 'b', 'c'], 'a')
```
